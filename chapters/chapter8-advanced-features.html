Master advanced ECharts functionality for professional visualizations

8.1 Data Zoom & Panning

Data zoom allows users to zoom in on specific data ranges and pan through large datasets. This is essential for handling large amounts of data effectively.

Interactive Data Zoom Example


var dataZoomChart = echarts.init(document.getElementById('data-zoom-chart'));
var dataZoomOption = {
    title: { text: 'Large Dataset with Data Zoom', left: 'center' },
    tooltip: { trigger: 'axis' },
    dataZoom: [
        {
            type: 'slider',
            show: true,
            xAxisIndex: [0],
            start: 0,
            end: 50
        },
        {
            type: 'inside',
            xAxisIndex: [0],
            start: 0,
            end: 50
        }
    ],
    xAxis: {
        type: 'category',
        data: Array.from({length: 1000}, (_, i) => `Day ${i+1}`)
    },
    yAxis: { type: 'value' },
    series: [{
        name: 'Sales',
        type: 'line',
        data: Array.from({length: 1000}, () => Math.random() * 100),
        smooth: true
    }]
};
dataZoomChart.setOption(dataZoomOption);

8.2 Advanced Animations

ECharts provides powerful animation capabilities that can make your charts more engaging and help users understand data changes over time.

Animated Chart with Progressive Loading


var animatedChart = echarts.init(document.getElementById('animated-chart'));
var animatedOption = {
    title: { text: 'Progressive Animation', left: 'center' },
    tooltip: { trigger: 'axis' },
    animation: true,
    animationDuration: 2000,
    animationEasing: 'cubicOut',
    animationDelay: function (idx) {
        return idx * 100;
    },
    xAxis: { data: ['Jan', 'Feb', 'Mar', 'Apr', 'May', 'Jun'] },
    yAxis: {},
    series: [{
        name: 'Revenue',
        type: 'bar',
        data: [120, 200, 150, 80, 70, 110],
        itemStyle: {
            color: new echarts.graphic.LinearGradient(0, 0, 0, 1, [
                { offset: 0, color: '#83bff6' },
                { offset: 0.5, color: '#188df0' },
                { offset: 1, color: '#188df0' }
            ])
        }
    }]
};
animatedChart.setOption(animatedOption);

8.3 Custom Themes

Create consistent visual styles across your application with custom themes. ECharts themes define colors, fonts, and other visual properties.

Custom Dark Theme


// Custom dark theme
var darkTheme = {
    backgroundColor: '#1a1a1a',
    textStyle: { color: '#fff' },
    title: { textStyle: { color: '#fff' } },
    legend: { textStyle: { color: '#fff' } },
    xAxis: { 
        axisLine: { lineStyle: { color: '#333' } },
        axisLabel: { color: '#fff' }
    },
    yAxis: { 
        axisLine: { lineStyle: { color: '#333' } },
        axisLabel: { color: '#fff' }
    }
};

var darkThemeChart = echarts.init(document.getElementById('dark-theme-chart'));
var darkOption = {
    title: { text: 'Dark Theme Chart', left: 'center' },
    tooltip: { trigger: 'axis' },
    legend: { data: ['Series A', 'Series B'] },
    xAxis: { data: ['Mon', 'Tue', 'Wed', 'Thu', 'Fri'] },
    yAxis: {},
    series: [
        {
            name: 'Series A',
            type: 'line',
            data: [120, 132, 101, 134, 90],
            itemStyle: { color: '#ff6b6b' }
        },
        {
            name: 'Series B',
            type: 'line',
            data: [220, 182, 191, 234, 290],
            itemStyle: { color: '#4ecdc4' }
        }
    ]
};
darkThemeChart.setOption(darkOption);

8.4 Advanced Tooltips

Custom tooltips can provide rich, contextual information and enhance user interaction with your charts.

Custom Tooltip with HTML


var customTooltipChart = echarts.init(document.getElementById('custom-tooltip-chart'));
var customTooltipOption = {
    title: { text: 'Custom Tooltip Example', left: 'center' },
    tooltip: {
        trigger: 'axis',
        formatter: function(params) {
            var result = '
';
            result += '
' + params[0].name + '
';
            params.forEach(function(param) {
                result += '
';
                result += '';
                result += param.seriesName + ': ' + param.value + '';
                result += '
';
            });
            result += '
';
            return result;
        }
    },
    xAxis: { data: ['Q1', 'Q2', 'Q3', 'Q4'] },
    yAxis: {},
    series: [
        {
            name: 'Revenue',
            type: 'bar',
            data: [120, 200, 150, 80],
            itemStyle: { color: '#4ecdc4' }
        },
        {
            name: 'Profit',
            type: 'bar',
            data: [20, 40, 30, 15],
            itemStyle: { color: '#ff6b6b' }
        }
    ]
};
customTooltipChart.setOption(customTooltipOption);

8.5 Dynamic Data Updates

Learn how to update chart data dynamically for real-time applications and interactive dashboards.

Real-time Data Updates


var dynamicChart = echarts.init(document.getElementById('dynamic-chart'));
var dynamicOption = {
    title: { text: 'Dynamic Data Updates', left: 'center' },
    tooltip: { trigger: 'axis' },
    xAxis: { data: ['A', 'B', 'C', 'D', 'E'] },
    yAxis: {},
    series: [{
        name: 'Values',
        type: 'line',
        data: [10, 20, 15, 25, 18],
        smooth: true
    }]
};
dynamicChart.setOption(dynamicOption);

function updateData() {
    var newData = Array.from({length: 5}, () => Math.floor(Math.random() * 30) + 5);
    dynamicChart.setOption({
        series: [{
            data: newData
        }]
    });
}

üí° Advanced Features Best Practices

  • Data Zoom: Use for datasets with more than 100 data points
  • Animations: Keep animations under 2 seconds for better UX
  • Themes: Create consistent themes across your application
  • Tooltips: Provide context and actionable information
  • Performance: Use data sampling for very large datasets

8.6 Key Takeaways

  • Data zoom is essential for large datasets and user exploration
  • Animations enhance user experience but should be used judiciously
  • Custom themes ensure visual consistency across applications
  • Advanced tooltips provide rich contextual information
  • Dynamic updates enable real-time data visualization
