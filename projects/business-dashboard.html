// Dashboard initialization
document.addEventListener('DOMContentLoaded', function() {
    // Initialize all charts
    const revenueChart = echarts.init(document.getElementById('revenue-chart'));
    const categoryChart = echarts.init(document.getElementById('category-chart'));
    const regionalChart = echarts.init(document.getElementById('regional-chart'));
    const demographicsChart = echarts.init(document.getElementById('demographics-chart'));
    const performanceChart = echarts.init(document.getElementById('performance-chart'));
    
    // Set options for each chart
    revenueChart.setOption(revenueOption);
    categoryChart.setOption(categoryOption);
    regionalChart.setOption(regionalOption);
    demographicsChart.setOption(demographicsOption);
    performanceChart.setOption(performanceOption);
    
    // Handle window resize
    window.addEventListener('resize', function() {
        revenueChart.resize();
        categoryChart.resize();
        regionalChart.resize();
        demographicsChart.resize();
        performanceChart.resize();
    });
});
